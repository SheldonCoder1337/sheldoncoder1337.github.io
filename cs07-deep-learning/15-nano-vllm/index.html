    Nano-vLLM

    Overview

    example.py
    import os
from nanovllm import LLM, SamplingParams
from transformers import AutoTokenizer


def main():
    path = os.path.expanduser("~/huggingface/Qwen3-0.6B/")

    # Step 1: 初始化 LLM 引擎
    llm = LLM(path, enforce_eager=True, tensor_parallel_size=1) # (1)

    # Step 2: 配置生成参数并准备 prompts
    sampling_params = SamplingParams(temperature=0.6, max_tokens=256)
    prompts = [
        "introduce yourself",
        "list all prime numbers within 100",
    ]
    tokenizer = AutoTokenizer.from_pretrained(path)
    prompts = [
        tokenizer.apply_chat_template(
            [{"role": "user", "content": prompt}],
            tokenize=False,
            add_generation_prompt=True,
        )
        for prompt in prompts
    ]

    # Step 3: 执行生成
    outputs = llm.generate(prompts, sampling_params) # (2)

    for prompt, output in zip(prompts, outputs):
        print("\n")
        print(f"Prompt: {prompt!r}")
        print(f"Completion: {output['text']!r}")

if __name__ == "__main__":
    main()

    2. nanovllm/engine/llm_engine.py
      class LLMEngine:
    def __init__(self, model, **kwargs):

        # 1. Config
        config = Config(model, **config_kwargs)

        # 2. ModelRunner
        self.model_runner = ModelRunner(config, 0, self.events)

        # 3. Tokenizer
        self.tokenizer = AutoTokenizer.from_pretrained(config.model, use_fast=True)
        config.eos = self.tokenizer.eos_token_id

        # 4. Scheduler
        self.scheduler = Scheduler(config)

    """ following topic will be covered later ...
    self.add_request -> self.scheduler.add(Sequence(prompt, sampling_params))
    self.is_finished -> self.scheduler.is_finished()
    self.step        -> self.scheduler.schedule()
                        self.model_runner.run()
                        self.scheduler.postprocess(seqs, token_ids)
    """

    def generate(self,
        prompts: list[str] | list[list[int]],
        sampling_params: SamplingParams | list[SamplingParams],
        use_tqdm: bool = True,
    ) -> list[str]:
        if use_tqdm:
            pbar = tqdm(total=len(prompts), desc="Generating", dynamic_ncols=True)
        if not isinstance(sampling_params, list):
            sampling_params = [sampling_params] * len(prompts)

        # step 1: 将每个请求包装为 Sequence 对象并加入调度器
        for prompt, sp in zip(prompts, sampling_params):
            self.add_request(prompt, sp)

        outputs = {}
        prefill_throughput = decode_throughput = 0.
        # step 2: 自回归循环，直到所有请求结束
        while not self.is_finished():
            t = perf_counter()
            # 核心：执行一步推理，包含调度、执行、后处理
            output, num_tokens = self.step()
            if use_tqdm:
                if num_tokens > 0:
                    prefill_throughput = num_tokens / (perf_counter() - t)
                else:
                    decode_throughput = -num_tokens / (perf_counter() - t)
                pbar.set_postfix({
                    "Prefill": f"{int(prefill_throughput)}tok/s",
                    "Decode": f"{int(decode_throughput)}tok/s",
                })
            for seq_id, token_ids in output:
                outputs[seq_id] = token_ids
                if use_tqdm:
                    pbar.update(1)
        # step 3: 收集结果并解码
        outputs = [outputs[seq_id] for seq_id in sorted(outputs.keys())]
        outputs = [{"text": self.tokenizer.decode(token_ids), "token_ids": token_ids} for token_ids in outputs]
        if use_tqdm:
            pbar.close()
        return outputs

    Takeaway msgs

    • 核心组件: Config、ModelRunner、Tokenizer、Scheduler、Sequence、SamplingParams。
    • 请求生命周期: add_request → scheduler.add(Sequence) → is_finished 检查 → 循环执行 step 直到完成。
    • step 内部流程:
      1. scheduler.schedule() 选出下一批待执行的序列。
      2. model_runner.run() 执行模型推理，返回新生成的 token IDs。
      3. scheduler.postprocess() 更新序列状态，标记完成的请求。

    Dive into LLMEngine

    nanovllm/engine/llm_engine.py
    import atexit
from dataclasses import fields
from time import perf_counter
from tqdm.auto import tqdm
from transformers import AutoTokenizer
import torch.multiprocessing as mp

from nanovllm.config import Config
from nanovllm.sampling_params import SamplingParams
from nanovllm.engine.sequence import Sequence
from nanovllm.engine.scheduler import Scheduler
from nanovllm.engine.model_runner import ModelRunner


class LLMEngine:

    def __init__(self, model, **kwargs):

        config_fields = {field.name for field in fields(Config)}
        config_kwargs = {k: v for k, v in kwargs.items() if k in config_fields}
        config = Config(model, **config_kwargs)

        self.ps = []
        self.events = []
        ctx = mp.get_context("spawn")
        for i in range(1, config.tensor_parallel_size):
            event = ctx.Event()
            process = ctx.Process(target=ModelRunner, args=(config, i, event))
            process.start()
            self.ps.append(process)
            self.events.append(event)
        self.model_runner = ModelRunner(config, 0, self.events)
        self.tokenizer = AutoTokenizer.from_pretrained(config.model, use_fast=True)
        config.eos = self.tokenizer.eos_token_id
        self.scheduler = Scheduler(config)
        atexit.register(self.exit)

    def exit(self):
        self.model_runner.call("exit")
        del self.model_runner
        for p in self.ps:
            p.join()

    def add_request(self, prompt: str | list[int], sampling_params: SamplingParams):
        if isinstance(prompt, str):
            prompt = self.tokenizer.encode(prompt)
        seq = Sequence(prompt, sampling_params)
        self.scheduler.add(seq)

    def step(self):
        seqs, is_prefill = self.scheduler.schedule()
        token_ids = self.model_runner.call("run", seqs, is_prefill)
        self.scheduler.postprocess(seqs, token_ids)
        outputs = [(seq.seq_id, seq.completion_token_ids) for seq in seqs if seq.is_finished]
        num_tokens = sum(len(seq) for seq in seqs) if is_prefill else -len(seqs)
        return outputs, num_tokens

    def is_finished(self):
        return self.scheduler.is_finished()

    def generate(
        self,
        prompts: list[str] | list[list[int]],
        sampling_params: SamplingParams | list[SamplingParams],
        use_tqdm: bool = True,
    ) -> list[str]:
        if use_tqdm:
            pbar = tqdm(total=len(prompts), desc="Generating", dynamic_ncols=True)
        if not isinstance(sampling_params, list):
            sampling_params = [sampling_params] * len(prompts)
        for prompt, sp in zip(prompts, sampling_params):
            self.add_request(prompt, sp)
        outputs = {}
        prefill_throughput = decode_throughput = 0.
        while not self.is_finished():
            t = perf_counter()
            output, num_tokens = self.step()
            if use_tqdm:
                if num_tokens > 0:
                    prefill_throughput = num_tokens / (perf_counter() - t)
                else:
                    decode_throughput = -num_tokens / (perf_counter() - t)
                pbar.set_postfix({
                    "Prefill": f"{int(prefill_throughput)}tok/s",
                    "Decode": f"{int(decode_throughput)}tok/s",
                })
            for seq_id, token_ids in output:
                outputs[seq_id] = token_ids
                if use_tqdm:
                    pbar.update(1)
        outputs = [outputs[seq_id] for seq_id in sorted(outputs.keys())]
        outputs = [{"text": self.tokenizer.decode(token_ids), "token_ids": token_ids} for token_ids in outputs]
        if use_tqdm:
            pbar.close()
        return outputs

    Dive into Scheduler

    调度器是整个系统的“大脑”，它决定了何时以及以何种顺序处理请求。其核心动机是： - 平衡吞吐量（每次处理更多请求）与延迟（尽快响应新请求）。 - 管理有限的 GPU 显存，特别是分页 KV 缓存（Paged KV Cache）。

    nanovllm/engine/scheduler.py
    from collections import deque

from nanovllm.config import Config
from nanovllm.engine.sequence import Sequence, SequenceStatus
from nanovllm.engine.block_manager import BlockManager


class Scheduler:

    def __init__(self, config: Config):
        self.max_num_seqs = config.max_num_seqs
        self.max_num_batched_tokens = config.max_num_batched_tokens
        self.eos = config.eos
        # BlockManager 管理分页 KV 缓存
        self.block_manager = BlockManager(config.num_kvcache_blocks, config.kvcache_block_size)
        self.waiting: deque[Sequence] = deque()   # 等待队列
        self.running: deque[Sequence] = deque()   # 运行队列

    def is_finished(self):
        return not self.waiting and not self.running

    def add(self, seq: Sequence):
        self.waiting.append(seq)

    def schedule(self) -> tuple[list[Sequence], bool]:
        # ---------- Prefill Phase ----------
        # 优先处理新请求的 Prefill（即首次处理 prompt）
        scheduled_seqs = []
        num_seqs = 0
        num_batched_tokens = 0
        while self.waiting and num_seqs < self.max_num_seqs:
            seq = self.waiting[0]
            # 检查是否满足 token 数限制和内存限制
            if num_batched_tokens + len(seq) > self.max_num_batched_tokens or not self.block_manager.can_allocate(seq):
                break
            num_seqs += 1
            self.block_manager.allocate(seq)
            num_batched_tokens += len(seq) - seq.num_cached_tokens   # 实际需要计算的新 token 数
            seq.status = SequenceStatus.RUNNING
            self.waiting.popleft()
            self.running.append(seq)
            scheduled_seqs.append(seq)
        if scheduled_seqs:
            return scheduled_seqs, True   # True 表示 prefill 阶段

        # ---------- Decode Phase ----------
        # 如果没有新请求，则处理已有的 decode 请求
        while self.running and num_seqs < self.max_num_seqs:
            seq = self.running.popleft()
            # 检查是否能分配新块来存放下一个 token
            while not self.block_manager.can_append(seq):
                # 如果当前序列无法扩展，尝试抢占（preempt）其他序列
                if self.running:
                    self.preempt(self.running.pop())
                else:
                    self.preempt(seq)
                    break
            else:
                num_seqs += 1
                self.block_manager.may_append(seq)
                scheduled_seqs.append(seq)
        assert scheduled_seqs
        self.running.extendleft(reversed(scheduled_seqs))
        return scheduled_seqs, False   # False 表示 decode 阶段

    def preempt(self, seq: Sequence):
        """抢占：将序列从运行队列移回等待队列，释放其占用的 KV 块"""
        seq.status = SequenceStatus.WAITING
        self.block_manager.deallocate(seq)
        self.waiting.appendleft(seq)

    def postprocess(self, seqs: list[Sequence], token_ids: list[int]) -> list[bool]:
        """更新序列状态，标记完成序列并释放资源"""
        for seq, token_id in zip(seqs, token_ids):
            seq.append_token(token_id)
            # 检查结束条件：遇到 EOS 或达到最大生成长度
            if (not seq.ignore_eos and token_id == self.eos) or seq.num_completion_tokens == seq.max_tokens:
                seq.status = SequenceStatus.FINISHED
                self.block_manager.deallocate(seq)
                self.running.remove(seq)

    Scheduler 关键点

    • 两阶段调度: schedule 优先处理 waiting 队列（Prefill），若没有则处理 running 队列（Decode）。
    • 内存感知: 通过 BlockManager 检查是否能为新序列分配足够块，或在 decode 阶段能否扩展最后一个块。
    • 抢占机制: 当内存不足时，将正在运行的序列放回等待队列，释放其 KV 块，为新请求腾出空间。

    Dive into ModelRunner

    ModelRunner 封装了模型执行的所有细节，包括多 GPU 并行、CUDA Graph 捕获、KV 缓存分配和输入预处理。其设计动机是： - 将复杂的模型推理逻辑与调度逻辑解耦。 - 利用 CUDA Graph 消除 kernel launch 开销，显著提升 decode 阶段性能。 - 支持 Tensor Parallelism，通过共享内存实现跨进程通信。

    nanovllm/engine/model_runner.py
    import pickle
import torch
import torch.distributed as dist
from multiprocessing.synchronize import Event
from multiprocessing.shared_memory import SharedMemory

from nanovllm.config import Config
from nanovllm.engine.sequence import Sequence
from nanovllm.models.qwen3 import Qwen3ForCausalLM
from nanovllm.layers.sampler import Sampler
from nanovllm.utils.context import set_context, get_context, reset_context
from nanovllm.utils.loader import load_model


class ModelRunner:

    def __init__(self, config: Config, rank: int, event: Event | list[Event]):
        self.config = config
        hf_config = config.hf_config
        self.block_size = config.kvcache_block_size
        self.enforce_eager = config.enforce_eager
        self.world_size = config.tensor_parallel_size
        self.rank = rank
        self.event = event

        # 初始化分布式环境
        dist.init_process_group("nccl", "tcp://localhost:2333", world_size=self.world_size, rank=rank)
        torch.cuda.set_device(rank)
        default_dtype = torch.get_default_dtype()
        torch.set_default_dtype(hf_config.torch_dtype)
        torch.set_default_device("cuda")

        # 加载模型
        self.model = Qwen3ForCausalLM(hf_config)
        load_model(self.model, config.model)
        self.sampler = Sampler()

        # 预热模型并分配 KV 缓存
        self.warmup_model()
        self.allocate_kv_cache()

        # 可选：捕获 CUDA Graph
        if not self.enforce_eager:
            self.capture_cudagraph()

        torch.set_default_device("cpu")
        torch.set_default_dtype(default_dtype)

        # 多卡情况下，非主卡进入循环监听共享内存
        if self.world_size > 1:
            if rank == 0:
                self.shm = SharedMemory(name="nanovllm", create=True, size=2**20)
                dist.barrier()
            else:
                dist.barrier()
                self.shm = SharedMemory(name="nanovllm")
                self.loop()   # 子进程进入无限循环，等待主进程指令

    def loop(self):
        """子进程循环：监听共享内存，执行远程方法调用"""
        while True:
            method_name, args = self.read_shm()
            self.call(method_name, *args)
            if method_name == "exit":
                break

    def call(self, method_name, *args):
        """统一调用入口，支持跨进程远程调用"""
        if self.world_size > 1 and self.rank == 0:
            self.write_shm(method_name, *args)
        method = getattr(self, method_name, None)
        return method(*args)

    def allocate_kv_cache(self):
        """根据 GPU 剩余显存自动计算可分配的 KV 块数量"""
        free, total = torch.cuda.mem_get_info()
        used = total - free
        peak = torch.cuda.memory_stats()["allocated_bytes.all.peak"]
        current = torch.cuda.memory_stats()["allocated_bytes.all.current"]
        num_kv_heads = hf_config.num_key_value_heads // self.world_size
        head_dim = getattr(hf_config, "head_dim", hf_config.hidden_size // hf_config.num_attention_heads)
        block_bytes = 2 * hf_config.num_hidden_layers * self.block_size * num_kv_heads * head_dim * hf_config.torch_dtype.itemsize
        config.num_kvcache_blocks = int(total * config.gpu_memory_utilization - used - peak + current) // block_bytes
        # 创建 KV 缓存张量，并分发给各层
        self.kv_cache = torch.empty(2, hf_config.num_hidden_layers, config.num_kvcache_blocks, self.block_size, num_kv_heads, head_dim)
        # ... 将切片赋值给每个 attention 层的 k_cache/v_cache ...

    @torch.inference_mode()
    def run_model(self, input_ids: torch.Tensor, positions: torch.Tensor, is_prefill: bool):
        """核心模型前向，根据阶段选择是否使用 CUDA Graph"""
        if is_prefill or self.enforce_eager or input_ids.size(0) > 512:
            return self.model.compute_logits(self.model(input_ids, positions))
        else:
            # 使用 CUDA Graph 执行 decode
            bs = input_ids.size(0)
            context = get_context()
            graph = self.graphs[next(x for x in self.graph_bs if x >= bs)]
            graph_vars = self.graph_vars
            # 更新 graph 输入
            graph_vars["input_ids"][:bs] = input_ids
            graph_vars["positions"][:bs] = positions
            graph_vars["slot_mapping"][:bs] = context.slot_mapping
            graph_vars["context_lens"][:bs] = context.context_lens
            graph_vars["block_tables"][:bs, :context.block_tables.size(1)] = context.block_tables
            graph.replay()
            return self.model.compute_logits(graph_vars["outputs"][:bs])

    def run(self, seqs: list[Sequence], is_prefill: bool) -> list[int]:
        """对外暴露的执行入口"""
        # 根据阶段准备输入
        input_ids, positions = self.prepare_prefill(seqs) if is_prefill else self.prepare_decode(seqs)
        temperatures = self.prepare_sample(seqs) if self.rank == 0 else None
        # 执行模型
        logits = self.run_model(input_ids, positions, is_prefill)
        # 采样生成下一个 token（仅 rank0 执行）
        token_ids = self.sampler(logits, temperatures).tolist() if self.rank == 0 else None
        reset_context()
        return token_ids

    ModelRunner 关键点

    • Tensor Parallelism 实现：主进程（rank0）通过 write_shm 将任务写入共享内存，子进程通过 loop 读取并执行，实现跨进程调用。
    • 自动 KV 缓存分配：allocate_kv_cache 根据当前 GPU 显存占用和配置的利用率，自动计算可容纳的 KV 块数量，无需手动调优。
    • CUDA Graph 优化：对于 decode 阶段且 batch size ≤ 512 的请求，使用预录制的 CUDA Graph 执行，大幅降低 kernel launch 开销。

    Dive into BlockManager

    在推理引擎中，KV 缓存（Key-Value Cache）的管理是影响吞吐量和内存效率的关键。Nano-vLLM 采用了分页 KV 缓存（Paged KV Cache） 技术，将每个序列的 KV 状态分割成固定大小的块（blocks），通过块表（block table）实现灵活的内存分配与共享。BlockManager 还在此基础上实现了前缀缓存（Prefix Caching）：当多个请求共享相同前缀时，自动复用已缓存的 KV 块，避免重复计算。

    nanovllm/engine/block_manager.py
    from collections import deque
import xxhash
import numpy as np

from nanovllm.engine.sequence import Sequence


class Block:
    """单个 KV 块的元数据"""
    def __init__(self, block_id):
        self.block_id = block_id
        self.ref_count = 0          # 引用计数，支持多序列共享
        self.hash = -1              # 块内容的哈希值，用于前缀匹配
        self.token_ids = []         # 块内 token 序列（仅用于前缀比较）


class BlockManager:

    def __init__(self, num_blocks: int, block_size: int):
        self.block_size = block_size
        self.blocks: list[Block] = [Block(i) for i in range(num_blocks)]
        self.hash_to_block_id: dict[int, int] = dict()   # 哈希 → 块 ID 映射
        self.free_block_ids: deque[int] = deque(range(num_blocks))  # 空闲块队列
        self.used_block_ids: set[int] = set()                       # 已分配块集合

    @classmethod
    def compute_hash(cls, token_ids: list[int], prefix: int = -1):
        """计算一个块的哈希值，可传入前缀哈希实现链式哈希"""
        h = xxhash.xxh64()
        if prefix != -1:
            h.update(prefix.to_bytes(8, "little"))
        h.update(np.array(token_ids).tobytes())
        return h.intdigest()

    def can_allocate(self, seq: Sequence) -> bool:
        """检查是否有足够空闲块来存放序列"""
        return len(self.free_block_ids) >= seq.num_blocks

    def allocate(self, seq: Sequence):
        """为序列分配块（Prefill 阶段调用）"""
        assert not seq.block_table
        h = -1
        cache_miss = False
        for i in range(seq.num_blocks):
            token_ids = seq.block(i)                     # 获取第 i 个块的内容
            # 如果块大小等于 block_size，计算其哈希（否则为 -1，表示不可缓存）
            h = self.compute_hash(token_ids, h) if len(token_ids) == self.block_size else -1
            block_id = self.hash_to_block_id.get(h, -1)  # 尝试从缓存中查找已有块
            # 若找到但内容不匹配（哈希冲突），则当作未命中
            if block_id == -1 or self.blocks[block_id].token_ids != token_ids:
                cache_miss = True
            if cache_miss:
                block_id = self.free_block_ids[0]        # 分配新块
                block = self._allocate_block(block_id)
            else:
                # 命中缓存：增加引用计数，更新序列已缓存 token 数
                seq.num_cached_tokens += self.block_size
                if block_id in self.used_block_ids:
                    block = self.blocks[block_id]
                    block.ref_count += 1
                else:
                    block = self._allocate_block(block_id)
            if h != -1:
                block.update(h, token_ids)
                self.hash_to_block_id[h] = block_id
            seq.block_table.append(block_id)

    def deallocate(self, seq: Sequence):
        """释放序列占用的所有块（序列完成或被抢占时调用）"""
        for block_id in reversed(seq.block_table):
            block = self.blocks[block_id]
            block.ref_count -= 1
            if block.ref_count == 0:
                self._deallocate_block(block_id)
        seq.num_cached_tokens = 0
        seq.block_table.clear()

    def can_append(self, seq: Sequence) -> bool:
        """检查 decode 阶段是否能给序列追加一个新 token"""
        return len(self.free_block_ids) >= (len(seq) % self.block_size == 1)

    def may_append(self, seq: Sequence):
        """decode 阶段，向序列追加一个 token（可能分配新块或更新最后块）"""
        block_table = seq.block_table
        last_block = self.blocks[block_table[-1]]
        if len(seq) % self.block_size == 1:
            # 当前 token 正好开始一个新块 → 分配新块
            assert last_block.hash != -1
            block_id = self.free_block_ids[0]
            self._allocate_block(block_id)
            block_table.append(block_id)
        elif len(seq) % self.block_size == 0:
            # 当前 token 刚好填满最后一个块 → 更新块的哈希信息
            assert last_block.hash == -1
            token_ids = seq.block(seq.num_blocks-1)
            prefix = self.blocks[block_table[-2]].hash if len(block_table) > 1 else -1
            h = self.compute_hash(token_ids, prefix)
            last_block.update(h, token_ids)
            self.hash_to_block_id[h] = last_block.block_id
        else:
            # 其他情况（最后一个块未满）无需操作
            assert last_block.hash == -1

    BlockManager 关键点

    • 分页管理：将 KV 缓存划分为固定大小的块（block_size），每个序列通过 block_table 记录其占用的块 ID。这种机制允许物理内存不连续，避免内存碎片。
    • 前缀缓存：通过 compute_hash 为每个块计算哈希值（支持链式哈希），并维护 hash_to_block_id 映射。当新请求的前缀与已缓存块匹配时，直接复用该块的 KV 状态，跳过该部分的前向计算。
    • 引用计数：每个块维护 ref_count，允许多个序列共享同一个块（例如共享 prompt 前缀）。当引用计数归零时，块被回收。
    • 动态扩展：decode 阶段每次生成一个 token，may_append 处理两种情况：若当前 token 是新块的开始，则分配新块；若刚好填满现有块，则更新块的哈希，使其后续可被其他请求复用。

    Dive into Sequence

    Sequence 类封装了单个生成请求的完整状态，包括 token 序列、采样参数、KV 块表以及生命周期状态。它是调度器和块管理器交互的基本单元。

    nanovllm/engine/sequence.py
    from copy import copy
from enum import Enum, auto
from itertools import count

from nanovllm.sampling_params import SamplingParams


class SequenceStatus(Enum):
    WAITING = auto()   # 等待调度
    RUNNING = auto()   # 正在执行
    FINISHED = auto()  # 已完成（EOS 或达到 max_tokens）


class Sequence:
    block_size = 256                      # 全局块大小，与 Config 一致
    counter = count()                     # 自动递增的序列 ID

    def __init__(self, token_ids: list[int], sampling_params = SamplingParams()):
        self.seq_id = next(Sequence.counter)
        self.status = SequenceStatus.WAITING
        self.token_ids = copy(token_ids)  # 完整 token 序列（prompt + 生成的 token）
        self.last_token = token_ids[-1]   # 最近一个 token（用于 decode 阶段输入）
        self.num_tokens = len(self.token_ids)
        self.num_prompt_tokens = len(token_ids)
        self.num_cached_tokens = 0        # 已缓存在 KV 中的 token 数
        self.block_table = []             # 该序列占用的块 ID 列表
        self.temperature = sampling_params.temperature
        self.max_tokens = sampling_params.max_tokens
        self.ignore_eos = sampling_params.ignore_eos

    @property
    def is_finished(self):
        return self.status == SequenceStatus.FINISHED

    @property
    def num_completion_tokens(self):
        return self.num_tokens - self.num_prompt_tokens

    @property
    def prompt_token_ids(self):
        return self.token_ids[:self.num_prompt_tokens]

    @property
    def completion_token_ids(self):
        return self.token_ids[self.num_prompt_tokens:]

    @property
    def num_cached_blocks(self):
        return self.num_cached_tokens // self.block_size

    @property
    def num_blocks(self):
        return (self.num_tokens + self.block_size - 1) // self.block_size

    @property
    def last_block_num_tokens(self):
        return self.num_tokens - (self.num_blocks - 1) * self.block_size

    def block(self, i):
        """返回第 i 个块内的 token 列表"""
        assert 0 <= i < self.num_blocks
        return self.token_ids[i*self.block_size: (i+1)*self.block_size]

    def append_token(self, token_id: int):
        self.token_ids.append(token_id)
        self.last_token = token_id
        self.num_tokens += 1

    # 序列化支持（用于跨进程传递）
    def __getstate__(self):
        return (self.num_tokens, self.num_prompt_tokens, self.num_cached_tokens, self.block_table,
                self.token_ids if self.num_completion_tokens == 0 else self.last_token)

    def __setstate__(self, state):
        self.num_tokens, self.num_prompt_tokens, self.num_cached_tokens, self.block_table = state[:-1]
        if self.num_completion_tokens == 0:
            self.token_ids = state[-1]
        else:
            self.last_token = state[-1]

    Sequence 关键点

    • 统一表示：将 prompt 和生成的 token 存储在同一个 token_ids 列表中，便于计算长度和切片。
    • 块级别操作：提供 block(i)、num_blocks、last_block_num_tokens 等属性，与 BlockManager 的块划分对齐。
    • 缓存状态追踪：num_cached_tokens 记录已缓存在 KV 中的 token 数，用于 prefill 阶段跳过已计算部分（前缀缓存优化）。
    • 序列化优化：当序列已生成部分 token 后，仅传递 last_token 和元数据，减少跨进程通信的数据量。

    Dive into Attention

    Nano-vLLM 的 Attention 层集成了 Flash Attention 的高效计算，并通过自定义 Triton kernel 将 KV 缓存写入分页内存。其核心设计动机是： - 利用 Flash Attention 的变长序列支持（flash_attn_varlen_func）高效处理 prefill 阶段。 - 在 decode 阶段使用 flash_attn_with_kvcache，直接从分页 KV 缓存读取。 - 通过 Triton kernel store_kvcache_kernel 实现非阻塞的 KV 缓存写入。

    nanovllm/layers/attention.py
    import torch
from torch import nn
import triton
import triton.language as tl

from flash_attn import flash_attn_varlen_func, flash_attn_with_kvcache
from nanovllm.utils.context import get_context


@triton.jit
def store_kvcache_kernel(
    key_ptr,
    key_stride,
    value_ptr,
    value_stride,
    k_cache_ptr,
    v_cache_ptr,
    slot_mapping_ptr,
    D: tl.constexpr,
):
    """将当前 step 计算出的 key/value 写入分页缓存中的指定位置"""
    idx = tl.program_id(0)
    slot = tl.load(slot_mapping_ptr + idx)
    if slot == -1: return
    key_offsets = idx * key_stride + tl.arange(0, D)
    value_offsets = idx * value_stride + tl.arange(0, D)
    key = tl.load(key_ptr + key_offsets)
    value = tl.load(value_ptr + value_offsets)
    cache_offsets = slot * D + tl.arange(0, D)
    tl.store(k_cache_ptr + cache_offsets, key)
    tl.store(v_cache_ptr + cache_offsets, value)


def store_kvcache(key: torch.Tensor, value: torch.Tensor, k_cache: torch.Tensor, v_cache: torch.Tensor, slot_mapping: torch.Tensor):
    """包装 kernel 调用，处理维度校验"""
    N, num_heads, head_dim = key.shape
    D = num_heads * head_dim
    # ... 确保内存布局连续 ...
    store_kvcache_kernel[(N,)](key, key.stride(0), value, value.stride(0), k_cache, v_cache, slot_mapping, D)


class Attention(nn.Module):

    def __init__(
        self,
        num_heads,
        head_dim,
        scale,
        num_kv_heads,
    ):
        super().__init__()
        self.num_heads = num_heads
        self.head_dim = head_dim
        self.scale = scale
        self.num_kv_heads = num_kv_heads
        self.k_cache = self.v_cache = torch.tensor([])   # 在 ModelRunner.allocate_kv_cache 中被赋值

    def forward(self, q: torch.Tensor, k: torch.Tensor, v: torch.Tensor):
        context = get_context()
        k_cache, v_cache = self.k_cache, self.v_cache

        # 如果缓存已存在，将当前 step 的 kv 写入缓存（prefill 或 decode 都会调用）
        if k_cache.numel() and v_cache.numel():
            store_kvcache(k, v, k_cache, v_cache, context.slot_mapping)

        if context.is_prefill:
            # Prefill 阶段：使用变长序列 Flash Attention
            if context.block_tables is not None:   # 若使用了前缀缓存，k/v 直接来自缓存
                k, v = k_cache, v_cache
            o = flash_attn_varlen_func(
                q, k, v,
                max_seqlen_q=context.max_seqlen_q, cu_seqlens_q=context.cu_seqlens_q,
                max_seqlen_k=context.max_seqlen_k, cu_seqlens_k=context.cu_seqlens_k,
                softmax_scale=self.scale, causal=True, block_table=context.block_tables
            )
        else:
            # Decode 阶段：使用带缓存读取的 Flash Attention
            o = flash_attn_with_kvcache(
                q.unsqueeze(1), k_cache, v_cache,
                cache_seqlens=context.context_lens, block_table=context.block_tables,
                softmax_scale=self.scale, causal=True
            )
        return o

    Attention 关键点

    • 双阶段处理：prefill 阶段处理整个 prompt，decode 阶段每次只生成一个 token，两者使用不同的 Flash Attention 接口。
    • 分页缓存写入：通过 store_kvcache_kernel 将计算出的 key/value 并行写入到 k_cache/v_cache 的指定 slot 位置。slot_mapping 由 ModelRunner 根据 block_table 和 num_cached_tokens 计算得出，确保写入正确位置。
    • 前缀缓存支持：当 context.block_tables 非空时，flash_attn_varlen_func 会使用 block_table 从分页缓存中读取 prefix 部分的 key/value，无需重复计算。

    Dive into Tensor Parallelism Layers

    为了支持多 GPU 推理，Nano-vLLM 实现了一套张量并行（Tensor Parallelism）的线性层。其核心思想是将权重矩阵按列或按行切分到不同 GPU，通过集合通信（all-reduce）聚合结果。

    nanovllm/layers/linear.py
    import torch
from torch import nn
import torch.nn.functional as F
import torch.distributed as dist


def divide(numerator, denominator):
    assert numerator % denominator == 0
    return numerator // denominator


class LinearBase(nn.Module):
    """所有并行线性层的基类，负责管理切分维度和权重加载器"""
    def __init__(self, input_size, output_size, bias=False, tp_dim=None):
        super().__init__()
        self.tp_dim = tp_dim
        self.tp_rank = dist.get_rank()
        self.tp_size = dist.get_world_size()
        self.weight = nn.Parameter(torch.empty(output_size, input_size))
        self.weight.weight_loader = self.weight_loader
        if bias:
            self.bias = nn.Parameter(torch.empty(output_size))
            self.bias.weight_loader = self.weight_loader
        else:
            self.register_parameter("bias", None)

    def weight_loader(self, param, loaded_weight):
        raise NotImplementedError


class ColumnParallelLinear(LinearBase):
    """按列切分：输出维度切分，输入完整复制"""
    def __init__(self, input_size, output_size, bias=False):
        tp_size = dist.get_world_size()
        super().__init__(input_size, divide(output_size, tp_size), bias, 0)

    def weight_loader(self, param, loaded_weight):
        param_data = param.data
        shard_size = param_data.size(self.tp_dim)
        start_idx = self.tp_rank * shard_size
        loaded_weight = loaded_weight.narrow(self.tp_dim, start_idx, shard_size)
        param_data.copy_(loaded_weight)

    def forward(self, x):
        return F.linear(x, self.weight, self.bias)


class RowParallelLinear(LinearBase):
    """按行切分：输入维度切分，输出后需 all-reduce"""
    def __init__(self, input_size, output_size, bias=False):
        tp_size = dist.get_world_size()
        super().__init__(divide(input_size, tp_size), output_size, bias, 1)

    def weight_loader(self, param, loaded_weight):
        param_data = param.data
        shard_size = param_data.size(self.tp_dim)
        start_idx = self.tp_rank * shard_size
        loaded_weight = loaded_weight.narrow(self.tp_dim, start_idx, shard_size)
        param_data.copy_(loaded_weight)

    def forward(self, x):
        y = F.linear(x, self.weight, self.bias if self.tp_rank == 0 else None)
        if self.tp_size > 1:
            dist.all_reduce(y)
        return y


class QKVParallelLinear(ColumnParallelLinear):
    """专门用于 QKV 投影：将 Q、K、V 的权重拼接后按列切分"""
    def __init__(self, hidden_size, head_size, total_num_heads, total_num_kv_heads=None, bias=False):
        tp_size = dist.get_world_size()
        total_num_kv_heads = total_num_kv_heads or total_num_heads
        self.head_size = head_size
        self.num_heads = divide(total_num_heads, tp_size)
        self.num_kv_heads = divide(total_num_kv_heads, tp_size)
        output_size = (total_num_heads + 2 * total_num_kv_heads) * self.head_size
        super().__init__(hidden_size, output_size, bias)

    def weight_loader(self, param, loaded_weight, loaded_shard_id):
        param_data = param.data
        if loaded_shard_id == "q":
            shard_size = self.num_heads * self.head_size
            shard_offset = 0
        elif loaded_shard_id == "k":
            shard_size = self.num_kv_heads * self.head_size
            shard_offset = self.num_heads * self.head_size
        else:  # "v"
            shard_size = self.num_kv_heads * self.head_size
            shard_offset = self.num_heads * self.head_size + self.num_kv_heads * self.head_size
        param_data = param_data.narrow(self.tp_dim, shard_offset, shard_size)
        loaded_weight = loaded_weight.chunk(self.tp_size, self.tp_dim)[self.tp_rank]
        param_data.copy_(loaded_weight)


class MergedColumnParallelLinear(ColumnParallelLinear):
    """合并多个线性层（如 gate 和 up）的列切分版本"""
    def __init__(self, input_size, output_sizes, bias=False):
        self.output_sizes = output_sizes
        super().__init__(input_size, sum(output_sizes), bias)

    def weight_loader(self, param, loaded_weight, loaded_shard_id):
        param_data = param.data
        shard_offset = sum(self.output_sizes[:loaded_shard_id]) // self.tp_size
        shard_size = self.output_sizes[loaded_shard_id] // self.tp_size
        param_data = param_data.narrow(self.tp_dim, shard_offset, shard_size)
        loaded_weight = loaded_weight.chunk(self.tp_size, self.tp_dim)[self.tp_rank]
        param_data.copy_(loaded_weight)

    Tensor Parallelism 关键点

    • 列并行（ColumnParallelLinear）：将输出维度切分，每个 GPU 持有完整输入权重的一部分。前向时直接计算，无需通信。
    • 行并行（RowParallelLinear）：将输入维度切分，每个 GPU 持有输出权重的完整副本，但只处理部分输入。计算完成后需执行 all-reduce 聚合结果。
    • 组合使用：Transformer 中，QKV 投影和 MLP 的 gate/up 投影通常使用列并行（可独立计算），而 down 投影使用行并行（需 all-reduce）。Attention 的 output 投影也是行并行。
    • 权重加载：通过 weight_loader 回调，从完整权重中提取当前 rank 对应的分片，支持从 safetensors 文件直接加载。

    Summary

    Nano-vLLM 的核心架构围绕“高吞吐、低延迟”的目标设计，每个组件都体现了极简且高效的理念：

    • LLMEngine：统一入口，协调调度与执行，管理请求生命周期。
    • Scheduler：基于内存感知的两阶段调度，优先处理新请求，通过抢占机制应对显存压力。
    • ModelRunner：封装多卡推理、CUDA Graph 和 KV 缓存分配，屏蔽底层复杂性。
    • BlockManager：分页 KV 缓存 + 前缀缓存，实现显存高效利用和计算复用。
    • Attention：结合 Flash Attention 与自定义 Triton kernel，实现高性能变长序列注意力。
    • Tensor Parallelism 层：通过简单的切分与集合通信，将模型扩展到多 GPU。

    整个项目代码量约 1200 行，却实现了与 vLLM 相近的性能，非常适合作为学习推理引擎实现的范本。
